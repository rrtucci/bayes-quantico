\chapter{$Spin(2)$}
\label{ch-spin-2}

Pauli matrices

\beq
\s_x=\s_1=\left(
\begin{array}{cc}
0&1
\\
1&0
\end{array}
\right),\quad
\s_y=\s_2=\left(
\begin{array}{cc}
0&-i
\\
i&0
\end{array}
\right),\quad
\s_z=\s_3=\left(
\begin{array}{cc}
1&0
\\
0&-1
\end{array}
\right)
\eeq

Hermitian and
unitary

\beq
\s_i^\dagger = \s_i,
\quad 
\s_i^2 = 1
\eeq


\beq
\s_x \s_y= -\s_y \s_x 
\eeq
\beq
\s_x \s_y = i \s_z
\eeq

 

\beq
\vec{\s}= (\s_1, \s_2, \s_3)
\eeq

\beq
\s_0=\left(
\begin{array}{cc}
1&0
\\
0&1
\end{array}
\right),\quad
\s_\mu = (\s_0, \vec{\s})
\eeq

\beq
e^{i\
\theta \s_i}=
\cos \theta
+
i \s_i \sin \theta 
\eeq
If $\vec{a}\in \RR^3$ then

\beq
e^{i\vec{a}\cdot\vec{\s}}=
\cos |\vec{a}|
+
i \frac{\vec{a}}{|\vec{a}|}\sin |\vec{a}|
\eeq

Define

\beq
Cl_2^+=
span\{1,i\s_3\}
=\{Ae^{i \theta \s_3}: \theta, A\in\RR\}
\cong\CC
\eeq

\beq
\vec{\s}_{1,2}=
(\s_1, \s_2)
\eeq
$Cl_2 =span(1, \vec{\s})$ is the 2-dim Clifford algebra
and $Cl_2^+=span(1, i\s_3=\s_1\s_2)$ 
is called the {\bf even Clifford subalgebra}
because it's generated by
products of an even number
of Pauli matrices.


\beq
\RR^2\cdot\vec{\s}_{12}=
\{
\vec{a}\cdot
\vec{\s}_{12}:
\vec{a}\in \RR^2
\}
\eeq

\beq
Spin(2)
=
\{
x\in Cl^+_2:
x x^\dagger=1,
xvx^\dagger
\in 
\RR^2\cdot\vec{\s}_{12} \text{ for all }
v\in \RR^2\cdot\vec{\s}_{12}\}
\eeq

\begin{claim}
\beq
Spin(2)=
\{
e^{i\theta\s_3}:
\theta\in\RR\}
\cong U(1)
\eeq
There is a 2 to 1 map
(i.e., a {\bf double cover map}) from $Spin(2)$
to $SO(2)$.
\end{claim}
\proof

Note that
if $x= e^{i\theta \s_3}$ and $v = v_1\s_1
+ v_2 \s_2$, 

\beqa
xx^\dagger &=&(\cos \theta + i\s_3 \sin \theta)(\cos \theta -i\s_3 \sin \theta)
\\
&=&
\cos^2\theta + \sin^2\theta 
\\
&=& 1
\eeqa

 Let 


\begin{align}
xvx^\dagger
&=
\left(
\begin{array}{cc}
e^{i\theta} & 0
\\
0 & e^{-i\theta}
\end{array}
\right)
\left(
\begin{array}{cc}
0 & v_1 - i v_2
\\
v_1 + i v_2 & 0
\end{array}
\right)
\left(
\begin{array}{cc}
e^{-i\theta} & 0
\\
0 & e^{i\theta}
\end{array}
\right)
\\
&=
\left(
\begin{array}{cc}
0&z^* e^{i2\theta}
\\
ze^{-i2\theta}&0
\end{array}
\right)  \quad (z = v_1 + i v_2)
\\
&=
\left(
\begin{array}{cc}
0 & v'_1 - i v'_2
\\
v'_1 + i v'_2 & 0
\end{array}
\right)
\end{align}

\beq
v_1' = \Re(ze^{-i2\theta})=
v_1 \cos 2\theta + v_2 \sin \theta
\eeq

\beq
v'_2 = \Im (ze^{-i2\theta})
=
-v_1 \sin 2\theta +
v_2 \cos 2\theta
\eeq

\beq
R_z(2\theta)=
\left(
\begin{array}{cc}
\cos 2\theta & \sin 2\theta
\\
-\sin 2\theta & \cos 2\theta
\end{array}
\right)
\eeq

\beq
\vec{v}= (v_1, v_2)^T,
\quad 
\vec{v'}= (v_1', v_2')^T
\eeq

\beq
\vec{v'}= R_z(2\theta)\vec{v}
\eeq

The 2-1 map:

\beq
e^{i\theta\s_3}\mapsto R_z(2\theta)
\eeq

\qed
