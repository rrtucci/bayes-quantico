\beq
\tr_{1,2,3}\cala_3(M^{\otimes 3})=
\frac{1}{3}
\left[
\tr(M) \tr_{1,2}\cala_2(M^{\otimes 2})
-\tr(M^2)\tr(M)
+\tr M^3
\right]
\eeq

\beq
\tr_{1,2}
\cala_2 M^{\otimes 2}=
\frac{1}{2}
\left[
\tr^2(M)-\tr(M^2)
\right]
\eeq

\beqa
\tr_{1,2,3}\cala_3(M^{\otimes 3})
&=&
\frac{1}{3}
\left[
\frac{1}{2}\tr^3(M) 
-\frac{3}{2}\tr(M^2)\tr(M)
+\tr M^3
\right]
\\
&=&
\frac{1}{3!}
\left[
\tr^3(M) 
-3\tr(M^2)\tr(M)
+2\tr M^3
\right]
\eeqa
