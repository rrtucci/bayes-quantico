Let $b^{nb}=(b_1, b_2, \ldots, b_{nb})$ where $b_i\in Z_{[0,nb_i]}$  and 
$a^{na}=(a_1, a_2, \ldots, a_{na})
$ where $a_i\in \ZZ_{[0,na_i]}$.
Hence,

\beq
d = \prod_{i=1}^{na}na_i, \;\;
d_\lam = \prod_{i=1}^{nb}nb_i
\eeq

\beq

\ecen
\eeq
