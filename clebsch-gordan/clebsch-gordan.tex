Let $b^{:nb}=(b_1, b_2, \ldots, b_{nb})$ where $b_i\in Z_{[0,db_i]}$  and $a\in Z_{[1,d_\lam]}$.
Hence,

\beq
d_\lam = \prod_{i=1}^{:nb}db_i
\eeq

\beq
(C_\lam)_{a}^{rev(b^{:nb})}=
\bcen
\xymatrix@C=1pc@R=1pc{
&&b_1\ar[dl]

}
\ecen
\eeq
