\begin{claim}

There is a 2 to 1 map
(i.e., a {\bf double cover map}) from $Spin(2)$
to $SO(2)$.
\end{claim}
\proof

Note $U= e^{i\theta \s_3}$ and $\vec{x}= (x_1, x_2)$

\begin{align}
U\s_{\vec{x}}U^\dagger
&=
\left(
\begin{array}{cc}
e^{i\theta} & 0
\\
0 & e^{-i\theta}
\end{array}
\right)
\left(
\begin{array}{cc}
0 & x_1 - i x_2
\\
x_1 + i x_2 & 0
\end{array}
\right)
\left(
\begin{array}{cc}
e^{-i\theta} & 0
\\
0 & e^{i\theta}
\end{array}
\right)
\\
&=
\left(
\begin{array}{cc}
0&z^* e^{i2\theta}
\\
ze^{-i2\theta}&0
\end{array}
\right)  \quad (z = x_1 + i x_2)
\\
&=
\left(
\begin{array}{cc}
0 & x'_1 - i x'_2
\\
x'_1 + i x'_2 & 0
\end{array}
\right)
\end{align}

\beq
x_1' = \Re(ze^{-i2\theta})=
x_1 \cos 2\theta + x_2 \sin \theta
\eeq

\beq
x'_2 = \Im (ze^{-i2\theta})
=
-x_1 \sin 2\theta +
x_2 \cos 2\theta
\eeq

\beq
R_z(2\theta)=
\left(
\begin{array}{cc}

\end{array}
\right)
\eeq

\beq

\quad 
\vec{x'}= (x_1', x_2')^T
\eeq

\beq
\vec{x'}= R_z(2\theta)\vec{x}
\eeq

The 2-1 map:

\beq
e^{i\theta\s_3}\mapsto R_z(2\theta)
\eeq

\qed

\section{Gammions}

\beq
a\cdot b= a_\mu b^\mu
\eeq

Instead of $\s\cdot\vec{x}=\s_{\vec {x}}$

\beq
\gamma\cdot a = \gamma_\mu a^\mu = \gamma_a
\eeq


\beq
[\gamma_\mu,\gamma_\nu]_+=2\eta_{\mu\nu}
\eeq

with $\eta_{\mu\nu}$ equals Euclidean $(1,1,1,1)$ or Lorentzian $(1, -1,-1,-1)$ metric.


\beq
\gamma_a \gamma_b
= a\cdot b + i\gamma_{\mu\nu} a^\mu b^\nu,
\eeq
where bivector

\beq
\gamma_{\mu\nu}=-i\frac{1}{2}[\gamma_\mu,\gamma_\nu]
\eeq
is the generator of $\mathfrak{so}(n)$ or $\mathfrak{so}(p,q)$.


A rotation by angle $\theta$ in 
direction $n^{\mu\nu}$ where 
$n^{\mu\nu}n_{\mu\nu}=1$ is

\beq
S(\theta)=e^{-i\frac{\theta}{2}n^{\mu\nu}\gamma_{\mu\nu}}
\eeq

\beq
 S \gamma_x S^\dagger = \gamma_{Rx}
\eeq
where $R\in SO(n)$ (or $SO(p,q)$).
 

\beq
Spin(n)=
\{e^{-i\frac{\theta}{2}n^{\mu\nu}\gamma_{\mu\nu}}|\theta\in \RR, 
n^{\mu\nu}n_{\mu\nu}=1
\}
\eeq

\beq
Cl(n)=span_\RR\{ \gamma_{a_1}\gamma_{a_2}\ldots \gamma_{a_{k}}| k=0,1,2, \ldots int(n/2)\}
\eeq

If
$n_-=n$ for $n$ even and $n_-=n-1$
for $n$ odd.

\beq
Cl^0(n) = span_\RR\{ \gamma_{a_1}\gamma_{a_2}\ldots \gamma_{a_{k}}| k=0,2,4, \ldots n_-\}
\eeq

\beq
Cl^1(n) = span_\RR\{ \gamma_{a_1}\gamma_{a_2}\ldots \gamma_{a_{k}}| k=1,3,5, \ldots n_-\}
\eeq

\beq
Cl(n)= Cl^0(n) \oplus Cl^1(n)
\eeq

Since $S$ is the exponentiation of a bivector, its Taylor expansion only
contains summands with an even number 
of gammas.
\beq
Spin(n)
\subset Cl^0(n)
\eeq


\beq
 - \gamma_{\hat{a}} \gamma_{x} \gamma_{\hat{a}}=
\gamma_{\rho_{\hat{a}}x}=\phi_{\hat{a}}(x)
\eeq

Reflection about axis $\hat{a}$
\beq
\rho_{\hat{a}}x
=x - 2(\hat{a}\cdot x)\hat{a}
\eeq

\beq
S^n =\{\hat{a}\in\RR^n| \hat{a}^2=1\}
\eeq

Pin stands for \qt{Product of involutions}
An involution in this case is a unit vector (i.e., $\hat{a}\in \RR^n$ such that
$\hat{a}\cdot\hat{a}=\hat{a}^2=1$)
\beqa
Pin(n)
&=&\cup_{k=1}^\infty
\{
\phi_{\hat{a}_{1}}
\circ
\phi_{\hat{a}_{1}}
\circ\ldots\circ
\phi_{\hat{a}_{k}}
| \hat{a}_{i}\in S^n \text{ for all $i$}\}
\eeqa

Note that

One refection
\beq
det[\phi_{\hat{a}}(x)] = -det(\gamma_x \gamma_{\hat{a}}^2)
=-\det(\gamma_{x})
\eeq

Two reflections = a rotation
\beq
det[\phi_{\hat{a}_1}\circ\phi_{\hat{a}_2}(x)] = +
\det(\gamma_{x})
\eeq

$Pin(n)$ is a double cover of $O(n)$
($det=\pm1$)
\beq
Pin(n)\xymatrix{\ar[r]_{\text{2 to 1 map}}&} O(n), \quad 
Pin(n)/\{1, -1\}\cong 
O(n)
\eeq

Another definition of $Spin(n)$
\beq
Spin(n)=\cup_{k=1}^\infty
\{
\phi_{\hat{a}_{1}}
\circ
\phi_{\hat{a}_{1}}
\circ\ldots\circ
\phi_{\hat{a}_{{2k}}}
| \hat{a}_{i}\in S^n \text{ for all $i$}\}
\eeq

$Spin(n)$ is a double cover of $SO(n)$
($\phi_{\hat{a}}=\phi_{-\hat{a}}
$)
\beq
Spin(n)\xymatrix{\ar[r]_{\text{2 to 1 map}}&} SO(n), \quad 
Spin(n)/\{1, -1\}\cong 
SO(n)
\eeq
