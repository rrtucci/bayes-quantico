Pauli matrices

\beq
\s_x=\left(
\begin{array}{cc}
0&1
\\
1&0
\end{array}
\right),\quad
\s_y=\left(
\begin{array}{cc}
0&-i
\\
i&0
\end{array}
\right),\quad
\s_z=\left(
\begin{array}{cc}
1&0
\\
0&-1
\end{array}
\right)
\eeq

\beq
\s_x \s_y= -\s_y \s_x 
\eeq
\beq
\s_x \s_y = i \s_z
\eeq

\begin{array}{cc}

\\

\end{array}
\right),

\eeq
