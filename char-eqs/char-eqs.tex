\beq
\tau = \tr(M)
\eeq


\beq
\tr_{1,2,3}\cala_3(M^{\otimes 3})=
\frac{1}{3}
\left[
\tau  \tr_{1,2}\cala_2(M^{\otimes 2})
-\tr(M^2)\tau 
+\tr M^3
\right]
\eeq

\beq
\tr_{1,2}
\cala_2 M^{\otimes 2}=
\frac{1}{2}
\left[
\tau^2-\tr(M^2)
\right]
\eeq

\beqa
\tr_{1,2,3}\cala_3(M^{\otimes 3})
&=&
\frac{1}{3}
\left[
\frac{1}{2}\tau^3 
-\frac{3}{2}\tr(M^2)\tau 
+\tr M^3
\right]
\\
&=&
\frac{1}{3!}
\left[
\tau^3
-3\tr(M^2)\tau 
+2\tr M^3
\right]
\eeqa

\beq
\tr_{1\ldots p}\cala_p M=
\frac{1}{p}
\sum_{k=1}^p
(-1)^{k-1}
\left(\tr_{1\ldots p-k}\cala_{p-k}M^{\otimes p-k}
\right)
\tr(M^k)
\eeq
