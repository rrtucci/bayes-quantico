\beq

\left(
\begin{array}{cc}

\eeq

\beq
f^Tf=I_{n},
\quad f^2 = 
-I_{n}
\eeq
$f^2$ is a primitive invariant matrix so it
must be proportional to
the identity. 

\beq
Sp(n)=\{G\in GL(n,\CC):
G^Tf G= f
\}
\eeq

\begin{claim}
If $G\in Sp(n)$, then $det(G)=1$.
\end{claim}
\proof

\beq
det(G^T f G)= det(f)
\eeq

\beq
det() = det(I_{n/2})det(-I_{n/2})= (-1)^{n/2}
\eeq

\beq
det(G^T f G) = det^2(G)
(-1)^{n/2}
\eeq


\beq
det^2(G)=1
\eeq

$det(G)=\pm 1$. $Sp(n)$
is connected, $I_n\in Sp(n)$  and $det(I_n)=1$.
Hence $det(G)=1$.

\qed
