\begin{claim}
\beq
\int dg\; G = 
\sum_{\lam\in \cals}
P_\lam
\eeq

\beq
\int dg\;
(G_\mu)_{ab}(G_\nu)^{cd}
=
\sum_{\lam \in \cals}
(P_\lam)\indices{_{ab}
^{cd}
}
\eeq

\end{claim}
\proof

\beqa
\int dg\; G
&=&
\sum_\lam
C^\dagger_\lam
\left[\int dg\; G_\lam
\right]
C_\lam
\\
&=& \sum_{\lam\in \cals}
C^\dagger_\lam C_\lam
\\
&=&
\sum_{\lam\in \cals}
P_\lam
\eeqa
