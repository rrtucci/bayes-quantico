\chapter{Integrals over a Group}
\label{ch-integrals}
This chapter is based on Cvitanovic's Birdtracks book Ref. \cite{birdtracks-book}.

In this chapter,
we will consider
only non-trivial reps;
i.e., reps with dimension $d_\lam>0$.
Trivial \qt{reps}, i.e., those with
$d_\lam=0$, will be excluded.

For a group $\calg$, let $SR=$ singlet reps
(i.e., one dimensional reps, scalar reps), and
$SR^c=$ nonsinglet reps.
Let $g$ be an
element of $\calg$
with a rep $G$.
The goal of this chapter is to show how to evaluate 
integrals over a group $\calg$, of the form:
\beq
\int dg 
G\indices{_a^b}
G\indices{_c^d}
\ldots
(G^\dagger)\indices{_e^f}
(G^\dagger)\indices{_g^h}
\eeq
subject to the 
constraints that:

\beq
\int dg =1
\eeq
and 

\beq
\int dg \;G_\lam
=0\quad  \text{if $\lam\in SR^c$}
\eeq

We will represent rep $G$ by
\beq
G\indices{_a^b}=
\xymatrix{a&G\ar@[green][l]&b\ar[l]},\quad
(G^\dagger)\indices{_b^a}=
\xymatrix{b&G^\dagger\ar@[green][l]&\ar[l]a}
\eeq
Note that we will always take the out arrow as the first one
(green).

We will assume that $G$
is a unitary matrix

\beq
\myboxed{
G^\dagger G=G G^\dagger =1
}
\quad 
\xymatrix@C=1.5pc{&G^\dagger
\ar[l]&G\ar[l]&\ar[l]}
=
\xymatrix@C=1.5pc{&G\ar[l]&G^\dagger\ar[l]&\ar[l]}=
\xymatrix@C=1.5pc{&\bullet\ar[l]&\ar[l]}
\eeq

Tensor products of $G$'s will be represented thus
\beq
G\otimes G\otimes G^\dagger=
\bcen
\xymatrix@R=1pc{
&\ar[l]G&\ar[l]
\\
&\ar[l]G&\ar[l]
\\
&\ar[l]G^\dagger&\ar[l]
}
\ecen
\eeq

\section{$\int\scriptstyle dg\; G$}
To evaluate $\int dg
\; G$, we 
expand $G$  in its
Clebsch-Gordan series. Such series and  the
Clebsch-Gordan 
coefficients $C_\lam$
are discussed in Chapter
\ref{ch-clebsch-gordan}.

\beqa
\int dg\; G
&=&
\sum_\lam
C^\dagger_\lam
\left[\int dg\; G_\lam
\right]
C_\lam
\\
&=& \sum_{\lam\in SR}
C^\dagger_\lam C_\lam
\\
&=&
\sum_{\lam\in SR}
P_\lam
\eeqa
This result is
valid for 
any group $\calg$
and any rep $G$
of that group.

\begin{claim}
For any group $\calg$
with reps $G_\mu$ and
$G_\nu$ (not necessarily irreps)

\beq
\int dg\;
(G_\mu)_{ab}(G_\nu)^{cd}
=
\sum_{\lam \in SR}
(P_\lam)\indices{_{ab}
^{cd}
}
\eeq

\end{claim}
\proof

Let
\beq
(C^\dagger_{\lam i})
\indices{
_{ac}
}
=
\bcen
\xymatrix@R=1pc{
\mu a&&&
\\
&C^\dagger_{\lam i}
\ar`u[ul][ul]
\ar`d[dl][dl]
&\ar[l]\lam i
\\
\nu c&&&
}
\ecen
\eeq
represent the
Clebsch-Gordan coefficients
for the Clebsch-Gordan
series $\mu\otimes \nu=\sum \lam$.

Since the $C_\lam$ are
invariant tensors:

\beq
\begin{array}{l}
\myboxed{
(G_\mu)\indices{_a^{a'}}
(G_\nu)\indices{
_{b'}^b}(C^\dagger_{\lam i})\indices{
_{a'}
^{b'}}
=
(C^\dagger_{\lam i'})\indices{_a^b}
(G_\lam)_{i'i}
}
\\
\bcen
\xymatrix@R=1pc{
&G_\mu\ar[l]&&
\\
&&C^\dagger_{\lam }
\ar`u[ul][ul]
\ar`d[dl][dl]
&\ar[l]
\\
&G_\nu\ar[l] &&
}
\ecen
=
\bcen
\xymatrix@R=1pc{
&&&
\\
&C^\dagger_{\lam }
\ar`u[ul][ul]
\ar`d[dl][dl]
&G_\lam\ar[l]
&\ar[l]
\\
&&&
}
\ecen
\end{array}
\eeq
Therefore,

\begin{align}
\int dg\;
\bcen
\xymatrix@R=1pc{
&G_\mu\ar[l]&\ar[l]
\\
&G_\nu\ar[l]&\ar[l]
}
\ecen
&=
\int dg\;
\sum_\lam
\bcen
\xymatrix@R=1pc{
&&&&
\ar`l[ld][ld]
\\
&C_{\lam}^\dagger
\ar`u[lu][lu]
\ar`d[dl][ld]
&G_\lam\ar[l]&\ar[l]
C_{\lam}
&
\\
&&&&\ar`l[ul][ul]
}
\ecen
\\
&=
\sum_{ij}
\sum_\lam
(C^\dagger_{\lam i})_{ab}
(C_{\lam j})^{cd}
\int dg
(G_{\lam })_{ij}
\\
&=
\sum_{i}
\sum_{\lam \in SR}
(C^\dagger_{\lam i})_{ab}
(C_{\lam i})^{cd}
\\
&=
\sum_{\lam \in SR}
(P_\lam)\indices{_{ab}
^{cd}}
\end{align}


\qed

\section{Character Orthonormality Relation}

For any matrix rep $G_\lam$ in rep $\lam$ such that $G_\lam$
represents the group
element $g$ in the 
Group $\calg$,
define the {\bf character 
of $g$
in rep $\lam$} by
\beq
\chi_\lam(g)=
\chi_\lam(G_\lam)
\eqdef
\tr G_\lam
=
(G_\lam)\indices{_a^a}
\eeq
Note that

\beq
\tr G^\dagger_\lam
=
(G\indices{_a^a})^*=
\chi_\lam(g)^*
\eeq

\begin{claim}
Suppose $G_\lam$ and $G_\mu$ 
are matrix reps
in irreps $\lam$ and $\mu$, respectively. Suppose
$h, G_\lam\in \CC^{d_\lam\times d_\lam}$
and $f, G_\mu\in \CC^{d_\mu\times d_\mu}$. Then
\beq
\begin{array}{l}
\myboxed{
\int dg\;
\chi_\lam(h^\dagger G_\lam)
\chi_\mu^*(G^\dagger_\mu f)
=
\delta(\mu, \lam)
\frac{1}{d_\mu}
\chi_\lam(h^\dagger f)
}
\\
\displaystyle
\int dg
\bcen
\xymatrix{
&h^\dagger\ar`r[rd]`[d][d]
&
\\
&G_\lam\ar`l[lu]`[u][u]
&
\\
&G^\dagger_\mu
\ar`r[dr]`[d][d]
&
\\
&f\ar`l[ul]`[u][u]
&
}
\ecen
=
\delta(\mu, \lam)
\frac{1}{d_\lam}
\bcen
\xymatrix{
&h^\dagger
\ar`r[dr]`[d][d]
&
\\
&f\ar`l[ul]`[u][u]
&
}
\ecen
\end{array}
\label{eq-char-ortho}
\eeq
\end{claim}
\proof

This claim follows from Eq.(\ref{eq-decomp-g-gdag} 
once we prove that
the left hand side of Eq.(\ref{eq-char-ortho})
is zero if $\lam\neq \mu$. Because $\lam$
and $\mu$ are both irreps, there can be
no matrix connecting 
$G_\mu$ and $G_\lam$
when $\lam\neq \mu$,
so the left hand side of Eq.(\ref{eq-char-ortho})
is indeed zero.
Even when $\lam=\mu$,
there can only be
one matrix,
namely
a Kronecker delta, connecting 
$G_\lam$ and $G_\mu$,
so the group must
have only one singlet rep.
\qed

Note that
since the matrices
$h, f\in \CC^{d_\mu\times d_\mu}$ are arbitrary,
differentiation can be used to retrieve 
$G_\mu$ from its character
with various $h$:
\beq
G\indices{_a^b}
=
\frac{d}{d(h^\dagger)
\indices{_b^a}
}
\underbrace{\chi_\mu(h^\dagger G)}
_{(h^\dagger)
\indices{_b^a}G\indices{_a^b}}
\eeq
