\section{$3j$ Coefficients}

Recall that $|\caly_\alp|$ or $|\alp|$
is the number of boxes (or number of outgoing legs in its birdtrack) in the YT $\caly_\alp$.  

\beq
\ger{C}_\beta=
\bcen
\xymatrix{
&&\ar@[green][dl]|\alp
\\
&\ar[l]|\beta \ger{C}_\beta&
\\
&&\ar[ul]|\gamma
}
\ecen
=
\bcen
\xymatrix@C=2pc{
&\caly_\beta
\ar@2{-}[d]
&\caly_\alp\ar[l]|{|\alp|}
&\ar[l]
\\
&\ar[l]|{|\beta|}&\caly_\gamma\ar[l]|{|\gamma|}&\ar[l]
}
\ecen
\eeq
where $|\beta| =
|\alp| +|\gamma|
$

\begin{claim}
For $U(n)$, we have\footnote{
Ref.\cite{birdtracks-book},
on which this claim
is based, uses different labels
for the Young projectors.
Their labels and ours are related as follows:
 
$\caly_\alpha\rarrow X$,
$\caly_\beta\rarrow  Y$,
$\caly_\gamma\rarrow  Z$.
} 
\beqa
\tr(\ger{C}^\dagger_\beta \ger{C}_\beta)=
\bcen
\xymatrix@C=3pc{
\ar@/^1.5pc/[r]|\alp
\ar@/_1.5pc/[r]|\gamma
\ger{C}_\beta^\dagger
&\ar[l]|\beta \ger{C}_\beta
}
\ecen
&=&
\bcen
\xymatrix@C=2pc{
&\caly_\beta
\ar@2{-}[d]\ar[l]
&\caly_\alp\ar[l]
&\ar@[red]@{-}@/_1pc/[lll]\ar[l]
\\
&\ar[l]&\caly_\gamma\ar[l]
&\ar@[red]@{-}@/_1pc/[lll]\ar[l]
}
\ecen
\\
&=& M dim(\caly_\beta)
\eeqa
where $M$ is independent of $n$.
\end{claim}
\proof

\beqa
\bcen
\xymatrix@C=2pc{
&\caly_\beta
\ar@2{-}[d]\ar[l]
&\caly_\alp\ar[l]
&\ar@[red]@{-}@/_1pc/[lll]\ar[l]
\\
&\ar[l]&\caly_\gamma\ar[l]
&\ar@[red]@{-}@/_1pc/[lll]\ar[l]
}
\ecen
&=&
\bcen
\xymatrix@C=2pc{
&\caly_\beta
\ar@2{-}[d]\ar[l]
&\caly_\alp\ar[l]
&\ar[l]
\caly_\beta\ar@2{-}[d]
&\ar@[red]@{-}@/_1pc/[llll]\ar[l]
\\
&\ar[l]&\caly_\gamma\ar[l]&\ar[l]
&\ar@[red]@{-}@/_1pc/[llll]\ar[l]
}
\ecen
\\
&=& M dim(\caly_\beta)
\eeqa
\qed

Ref. \cite{birdtracks-book}
shows that for this example, $M=1$.

\section{Sum Rules}

Let

$STY(n_b)=$ set of 
STY with $n_b$ boxes

$STY_+=$ set of STY with one or more boxes. Hence, SYT $\caly_\alp$ with $|\alp|=0$  is excluded.

\begin{claim}

\beq
\sum_{\alp, \gamma \in SY\ger{C}_+}\indi(|\alp|+|\gamma|=|\beta|)
\trij{\alp}{\beta}
{\gamma}
=
(|\beta|-1)dim(\caly_\beta)
\eeq

\end{claim}
\proof

\beq
\sum_{\alp\in SYT(|\alp_0|)}\caly_\alp = 1
,\quad
\sum_{\gamma\in SYT(|\gamma_0|)}\caly_\gamma = 1
\eeq

\begin{align}
\sum_{\alp, \gamma \in SY\ger{C}_+}
\indi(|\alp| + |\gamma| = |\beta|)
\trij{\alp}{\beta}
{\gamma}
&=
\sum_{|\alp|=1}^{|\beta|-1}
\begin{array}{c}
\sum
\\
\scriptstyle
\alp \in SYT(|\alp|)
\\
\scriptstyle
\gamma\in
SYT(|\beta|-|\alp| )
\end{array}
\bcen
\xymatrix@C=1pc{
&\caly_\beta
\ar@2{-}[d]\ar[l]
&\caly_\alp\ar[l]
&\ar@[red]@{-}@/_1pc/[lll]\ar[l]
\\
&\ar[l]&\caly_\gamma\ar[l]
&\ar@[red]@{-}@/_1pc/[lll]\ar[l]
}
\ecen
\\
&=
\sum_{|\alp|=1}^{|\beta|-1}
\bcen
\xymatrix@C=2pc{
&\caly_\beta
\ar@2{-}[d]\ar[l]
&\ar@[red]@{-}@/_1pc/[ll]\ar[l]
\\
&\ar[l]
&\ar@[red]@{-}@/_1pc/[ll]\ar[l]
}
\ecen
\\
&= (|\beta|-1)dim(\caly_\beta)
\end{align}

\qed

Let 

\beq
A = \{\rho, \nu, \lam, \s, \omega, \mu\},
\quad 
A(\omega)= A-\{\omega\}
\eeq

\beq
J(A)=
\indi
\left(
\begin{array}{l}
|\s|+|\mu|=|\omega|,
\\
|\nu|+|\rho|=|\omega|,
\\
|\s|+|\lam|=|\rho|,
\\
|\lam|+|\nu|=|\mu|
\end{array}
\right)
\eeq


\begin{claim}
\beq
\prod_{\alp \in A(\omega)}
\left[\sum_{\alp \in SYT(|\alp|)}\right]
J(A)
\bcen
\sixj{\lam}{\mu}{\nu}{\omega}{\rho}{\s}
\ecen
=
\frac{1}{2}
(|\omega|-1)(|\omega|-2)dim{\caly_\omega}
\eeq
\end{claim}
\proof

See Ref.\cite{birdtracks-book}
for proof.
\qed
