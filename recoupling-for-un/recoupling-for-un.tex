\section{$3-j$ coefficients}

\beq
T_\beta=
\bcen
\xymatrix{
&&\ar@[green][dl]|\alp
\\
&\ar[l]|\beta T_\beta&
\\
&&\ar[ul]|\gamma
}
\ecen
=
\bcen
\xymatrix@C=2pc{
&\caly_\beta
\ar@2{-}[d]
&\caly_\alp\ar[l]|{\ell_\alp}
&\ar[l]
\\
&\ar[l]|{\ell_\beta}&\caly_\gamma\ar[l]|{\ell_\gamma}&\ar[l]
}
\ecen
\eeq
where $\ell_\beta =
\ell_\alp +\ell_\gamma
$

\begin{claim}
\beqa
\tr(T^\dagger_\beta T_\beta)=
\bcen
\xymatrix@C=3pc{
\ar@/^1.5pc/[r]|\alp
\ar@/_1.5pc/[r]|\gamma
T_\beta^\dagger
&\ar[l]|\beta T_\beta
}
\ecen
&=&
\bcen
\xymatrix@C=2pc{
&\caly_\beta
\ar@2{-}[d]\ar[l]
&\caly_\alp\ar[l]
&\ar@[red]@{-}@/_1pc/[lll]\ar[l]
\\
&\ar[l]&\caly_\gamma\ar[l]
&\ar@[red]@{-}@/_1pc/[lll]\ar[l]
}
\ecen
\\
&=& M dim(\caly_\beta)
\eeqa
\end{claim}
\proof

\beqa
\bcen
\xymatrix@C=2pc{
&\caly_\beta
\ar@2{-}[d]\ar[l]
&\caly_\alp\ar[l]
&\ar@[red]@{-}@/_1pc/[lll]\ar[l]

&\ar[l]&\caly_\gamma\ar[l]
&\ar@[red]@{-}@/_1pc/[lll]\ar[l]
}
\ecen
&=&
\bcen
\xymatrix@C=2pc{

\ar@2{-}[d]\ar[l]

}
\ecen
\\
&=& M dim(\caly_\beta)
\eeqa
\qed

there are 12 Young projectors
on the LHS too.
