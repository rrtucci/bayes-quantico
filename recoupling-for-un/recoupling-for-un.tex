\chapter{Recoupling Identities for $U(n)$}
\label{ch-recoupling-for-un}

This chapter is based on Cvitanovic's Birdtracks book Ref. \cite{birdtracks-book}.

For $U(n)$ (as opposed to $SU(n)$), there are no antiparticles (i.e., one can use only lowered indices).
A consequence of this is  that for proper
operators in $U(n)$, the total particle number is conserved.

\section{$3j$ Coefficients}

Recall that $|\caly_\alp|$ or $|\alp|$
is the number of boxes (or number of outgoing legs in its birdtrack) in the YT $\caly_\alp$.  

Clebsch-Gordan (CG) coefficients are discussed in Chapter 
\ref{ch-clebsch-gordan}. One can
define a CG coefficient
$\ger{C}_\beta$ in terms of Young Tableau 
as follows:

\beq
\ger{C}_\beta=
\bcen
\xymatrix{
&&\ar@[green][dl]|\alp
\\
&\ar[l]|\beta \ger{C}_\beta&
\\
&&\ar[ul]|\gamma
}
\ecen
=
\bcen
\xymatrix@C=2pc{
&\caly_\beta
\ar@2{-}[d]
&\caly_\alp\ar[l]|{|\alp|}
&\ar[l]
\\
&\ar[l]|{|\beta|}&\caly_\gamma\ar[l]|{|\gamma|}&\ar[l]
}
\ecen
\eeq
where $|\beta| =
|\alp| +|\gamma|
$

\begin{claim}
($3j$ coefficient for $U(n)$ in terms of YT)\footnote{
Ref.\cite{birdtracks-book},
on which this claim
is based, uses different labels
for the Young projectors.
Their labels and ours are related as follows:
 
$\caly_\alpha\rarrow X$,
$\caly_\beta\rarrow  Y$,
$\caly_\gamma\rarrow  Z$.
} 
\beqa
\tr(\ger{C}^\dagger_\beta \ger{C}_\beta)=
\bcen
\xymatrix@C=3pc{
\ar@/^1.5pc/[r]|\alp
\ar@/_1.5pc/[r]|\gamma
\ger{C}_\beta^\dagger
&\ar[l]|\beta \ger{C}_\beta
}
\ecen
&=&
\bcen
\xymatrix@C=2pc{
&\caly_\beta
\ar@2{-}[d]\ar[l]
&\caly_\alp\ar[l]
&\ar@[red]@{-}@/_1pc/[lll]\ar[l]
\\
&\ar[l]&\caly_\gamma\ar[l]
&\ar@[red]@{-}@/_1pc/[lll]\ar[l]
}
\ecen
\\
&=& K dim(\caly_\beta)
\eeqa
where $K $ is independent of $n$.
\end{claim}
\proof

\beqa
\bcen
\xymatrix@C=2pc{
&\caly_\beta
\ar@2{-}[d]\ar[l]
&\caly_\alp\ar[l]
&\ar@[red]@{-}@/_1pc/[lll]\ar[l]
\\
&\ar[l]&\caly_\gamma\ar[l]
&\ar@[red]@{-}@/_1pc/[lll]\ar[l]
}
\ecen
&=&
\bcen
\xymatrix@C=2pc{
&\caly_\beta
\ar@2{-}[d]\ar[l]
&\caly_\alp\ar[l]
&\ar[l]
\caly_\beta\ar@2{-}[d]
&\ar@[red]@{-}@/_1pc/[llll]\ar[l]
\\
&\ar[l]&\caly_\gamma\ar[l]&\ar[l]
&\ar@[red]@{-}@/_1pc/[llll]\ar[l]
}
\ecen
\\
&=& K dim(\caly_\beta)
\eeqa
\qed

Ref. \cite{birdtracks-book}
shows that for this example, $K=1$.

($6j$ coefficient for $U(n)$ in terms of YT)\footnote{
Ref.\cite{birdtracks-book},
on which this claim
is based, uses different labels
for the Young projectors.
Their labels and ours are related as follows:
 
$\caly_\s\rarrow X$,
$\caly_\omega\rarrow  Y$,
$\caly_\rho\rarrow  U$,
$\caly_\lam\rarrow W$,
$\caly_\nu\rarrow V$,
$\caly_\mu\rarrow Z$.
}
\beqa
\bcen
\sixj{\lam}{\mu}{\nu}{\omega}{\rho}{\s}
\ecen
&=&
\bcen
\xymatrix@C=1pc@R=1pc{
&
&\caly_\nu\ar[ll]
&\caly_\mu\ar@2{-}[d]
\ar[l]

K dim(\caly_\omega)
\eeqa
where $K$ is independent of $n$
\end{claim}

12 Young projectors on the RHS.

on the left hand side (LHS) is composed of
3 Young projectors  so
there are 12 Young projectors
on the LHS too.

\section{Sum Rules}

Let

$SYT(n_b)=$ set of 
SYT with $n_b$ boxes

$SYT=\bigcup_{n_b=1}^{\infty} SYT(n_b)$ 


\begin{claim}

\beq
\sum_{\alp', \gamma' \in SYT}\indi(|\alp'|+|\gamma'|=|\beta|)
\trij{\alp'}{\beta}
{\gamma'}
=
(|\beta|-1)dim(\caly_\beta)
\eeq

\end{claim}
\proof

\beq
\sum_{\alp'\in SYT(|\alp|)}\caly_{\alp'} = 1
,\quad
\sum_{\gamma'\in SYT(|\gamma|)}\caly_{\gamma'} = 1
\eeq

\begin{align}
\sum_{{\alp'}, {\gamma'} \in SYT}
\indi(|{\alp'}| + |{\gamma'}| = |\beta|)
\trij{{\alp'}}{\beta}
{{\gamma'}}
&=
\sum_{|{\alp'}|=1}^{|\beta|-1}
\begin{array}{c}
\sum
\\
\scriptstyle
{\alp'} \in SYT(|{\alp'}|)
\\
\scriptstyle
{\gamma'}\in
SYT(|\beta|-|{\alp'}| )
\end{array}
\bcen
\xymatrix@C=1pc{
&\caly_\beta
\ar@2{-}[d]\ar[l]
&\caly_{\alp'}\ar[l]
&\ar@[red]@{-}@/_1pc/[lll]\ar[l]
\\
&\ar[l]&\caly_{\gamma'}\ar[l]
&\ar@[red]@{-}@/_1pc/[lll]\ar[l]
}
\ecen
\\
&=
\sum_{|{\alp'}|=1}^{|\beta|-1}
\bcen
\xymatrix@C=2pc{
&\caly_\beta
\ar@2{-}[d]\ar[l]
&\ar@[red]@{-}@/_1pc/[ll]\ar[l]
\\
&\ar[l]
&\ar@[red]@{-}@/_1pc/[ll]\ar[l]
}
\ecen
\\
&= (|\beta|-1)dim(\caly_\beta)
\end{align}

\qed

Let 

\beq
A = \{\rho, \nu, \lam, \s, \omega, \mu\},
\quad 
A_-= A-\{\omega\}
\eeq
\beq
A' = \{\rho', \nu', \lam', \s', \omega', \mu'\},\quad
A'_-= A' -\{\omega'\}
\eeq

\beq
J(A)=
\indi
\left(
\begin{array}{l}
|\s|+|\mu|=|\omega|,
\\
|\nu|+|\rho|=|\omega|,
\\
|\s|+|\lam|=|\rho|,
\\
|\lam|+|\nu|=|\mu|
\end{array}
\right)
\eeq


\begin{claim}
\beq
\prod_{\alp' \in A_-'}
\left[\sum_{\alp' \in SYT}\right]
J(A')
\bcen
\sixj{\lam'}{\mu'}{\nu'}{\omega}{\rho'}{\s'}
\ecen
=
\frac{1}{2}
(|\omega|-1)(|\omega|-2)dim{\caly_\omega}
\eeq
\end{claim}
\proof

See Ref.\cite{birdtracks-book}
for proof.
\qed
